import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.StringTokenizer;
import java.io.BufferedReader;
import java.io.BufferedWriter;

class Main {
    private static int n, m;
    private static int[][] arr;

    public static boolean Full(int x, int y, int size) {
        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                if (arr[x + i][y + j] == 0) {
                    return false;
                }
            }
        }
        return true;
    }

    public static int Check(int[][] arr) {
        int N = Math.min(n, m);
        for (int size = N; size > 1; size--) {
            for (int x = 0; x + size < n; x++) {
                for (int y = 0; y + size < m; y++) {
                    if (Full(x, y, size)) {
                        return size * size;
                    }
                }
            }
        }
        return 1;
    }

    public static void main(String args[]) throws Exception {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        arr = new int[n][m];
        for (int i = 0; i < n; i++) {
            String str = br.readLine();
            for (int j = 0; j < m; j++) {
                arr[i][j] = str.charAt(j) - '0';
            }
        }
        bw.write(Check(arr) + "\n");
        bw.close();
        br.close();
    }
}

import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.util.StringTokenizer;
import java.io.BufferedReader;
import java.io.BufferedWriter;

class Main {
    private static int n, m, Max;
    private static int[][] arr;

    public static void dp(int[][] arr) {
        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= m; j++) {
                if (arr[i][j] == 1) {
                    arr[i][j] = Math.min(Math.min(arr[i - 1][j], arr[i][j - 1]) + 1, arr[i - 1][j - 1] + 1);
                    Max = (Max < arr[i][j]) ? arr[i][j] : Max;
                }
            }
        }
    }

    public static void main(String args[]) throws Exception {
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine(), " ");
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        Max = 0;
        arr = new int[n + 1][m + 1];
        for (int i = 1; i <= n; i++) {
            String str = br.readLine();
            for (int j = 1; j <= m; j++) {
                arr[i][j] = str.charAt(j - 1) - '0';
            }
        }
        dp(arr);
        bw.write(Max * Max + "\n");
        bw.close();
        br.close();
    }
}
