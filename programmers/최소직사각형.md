class Solution {
    public int solution(int[][] sizes) {
        int w = 0, h = 0, min = 0, max = 0;
        for(int[] size : sizes) {
            min = (size[0] < size[1]) ? size[0] : size[1];
            max = (size[0] < size[1]) ? size[1] : size[0];
            w = (w < max) ? max : w;
            h = (h < min) ? min : h;
        }
        return w * h;
    }
}
