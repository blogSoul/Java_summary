import java.util.*;

class Solution {
    public int solution(int[] people, int limit) {
        int answer = 0, start = 0, end = people.length - 1;
        Arrays.sort(people);
        while(start <= end){
            int a = people[start];
            int b = people[end];
            if(start == end){
                answer += 1;
                break;
            }
            if(a + b > limit){
                end -= 1;
                answer += 1;
            } else {
                start += 1;
                end -= 1;
                answer += 1;
            }
        }
        return answer;
    }
}
