class Solution {
    public boolean isComplete(int num, int[] stones, int k){
        int count = 0;
        for(Integer i : stones){
            if(i - num <= 0){
                count++;
            } else {
                count = 0;
            }
            if(count == k){
                return false;
            }
        }
        return true;
    }
    public int solution(int[] stones, int k) {
        int answer = 0;
        int Min = Integer.MAX_VALUE, Max = Integer.MIN_VALUE;
        for(Integer i : stones){
            Min = Math.min(Min, i);
            Max = Math.max(Max, i);
        }
        int i = Min, j = Max;
        while(i < j){
            int mid = (i + j) / 2;
            if(isComplete(mid, stones, k)){
                i = mid + 1;
            } else {
                j = mid;
            }
        }
        return j;
    }
}

class Solution {
    public int[] stones;
    public static int k, answer;
    public int check(int[] stones, int k) {
        int Size = stones.length;
        int start = -1;
        for(int i = 0 ; i < k; i++){
            if(stones[i] > 0){
                start = i;
                break;
            }
        }
        if(start == -1){
            return -1;
        } // to cal start index
        int step_size = Integer.MAX_VALUE;
        boolean isCompelete = false;
        boolean[] visited = new boolean[Size];
        while(start < Size){
            isCompelete = false;
            step_size = Math.min(step_size, stones[start]);
            for(int step = 1; step <= k; step++){
                if(step + start >= Size){
                    isCompelete = true;
                    visited[start] = true;
                    start += step;
                    break;
                }
                if(stones[step + start] > 0){
                    isCompelete = true;
                    visited[start] = true;
                    start += step;
                    break;
                }
            }
            if(isCompelete){
                continue;
            } else {
                return -1;
            }
        }
        for(int i = 0 ; i < Size; i++){
            if(visited[i]){
                stones[i] -= step_size;
            }
        }
        answer += step_size;
        return step_size;
    }
    public int solution(int[] stones, int k) {
        answer = 0;
        while(check(stones, k) != -1){
            continue;
        }
        return answer;
    }
}
