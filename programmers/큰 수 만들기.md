class Solution {
    public void dfs(int count, String number, boolean[] visited, char[] result){
        if(count == result.length){
            for(char i : result){
                System.out.print(i + " ");
            }
            System.out.print("\n");
            return;
        }
        int size = visited.length;
        for(int i = 0; i < size; i++){
            if(!visited[i]){
                visited[i] = true;
                result[count] = number.charAt(i);
                dfs(count + 1, number, visited, result);
                visited[i] = false;
            }
        }
    }
    public String solution(String number, int k) {
        String answer = "";
        int number_size = number.length(), size = number_size - k;
        boolean[] visited = new boolean[number_size];
        char[] result = new char[size];
        dfs(0, number, visited, result);
        return answer;
    }
}
